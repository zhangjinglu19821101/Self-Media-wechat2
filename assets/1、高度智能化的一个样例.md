/**
 * 核心类型定义：贴合保险领域合规校验场景

// MCP接口文档结构化模型（Agent B参考的依据）
export interface MCPAPIDoc {
  apiName: string; // 接口名称（如"保险科普文发布校验"）
  complianceRules: string[]; // 合规规则（自然语言描述，交给LLM解析）
  requiredFields: string[]; // 必填字段（基础校验，硬编码兜底）
  fieldFormatRules: Record<string, string>; // 字段格式规则（自然语言）
  errorCodeMapping: Record<string, string>; // 错误码映射（基础兜底）
}

// 执行Agent（如insurance-d）返回内容
export interface AgentResponse {
  content: Record<string, any>; // 核心返回内容
  domainScene: string; // 保险领域场景（如"医疗险科普-公众号发布"）
  capabilityType: string; // MCP能力类型
}

// 合规校验结果（Agent B输出的凭证）
export interface ComplianceCheckResult {
  pass: boolean; // 是否合规
  checkDetails: string; // 智能校验详情（LLM生成）
  violationRules?: string[]; // 违规规则（LLM识别）
  correctionSuggestion: string; // 智能整改建议（Agent B决策）
  credential: string; // 合规凭证编号（自动生成）
}

// LLM调用配置
export interface LLMConfig {
  apiUrl: string;
  apiKey: string;
  model: string; // 如gpt-4o/claude-3
}
3. 核心合规校验服务（ComplianceCheckService.ts）
typescript
运行
import axios from 'axios';
import { z } from 'zod';
import {
  MCPAPIDoc,
  AgentResponse,
  ComplianceCheckResult,
  LLMConfig
} from './types';
import dotenv from 'dotenv';

dotenv.config();

/**
 * MCP合规校验智能化服务
 * 核心逻辑：Agent B（技术总监）基于接口文档+Agent返回内容，靠LLM做智能校验
 * 仅保留"必填字段存在性"硬编码，其余合规规则全由LLM推理
 */
export class ComplianceCheckService {
  private llmConfig: LLMConfig;

  constructor() {
    // 初始化LLM配置（从环境变量读取，硬编码仅做配置读取）
    this.llmConfig = {
      apiUrl: process.env.LLM_API_URL!,
      apiKey: process.env.LLM_API_KEY!,
      model: process.env.LLM_MODEL || 'gpt-4o'
    };
  }

  /**
   * 核心方法：智能化合规校验
   * @param apiDoc MCP接口文档（Agent B参考的依据）
   * @param agentResponse 执行Agent返回内容
   * @returns 合规校验结果（含Agent B生成的凭证）
   */
  async checkCompliance(
    apiDoc: MCPAPIDoc,
    agentResponse: AgentResponse
  ): Promise<ComplianceCheckResult> {
    try {
      // 步骤1：基础必填字段校验（仅有的硬编码兜底）
      const basicCheckPass = this.checkRequiredFields(apiDoc, agentResponse.content);
      const basicCheckMsg = basicCheckPass 
        ? "必填字段校验通过" 
        : `必填字段缺失：${this.getMissingFields(apiDoc, agentResponse.content).join(', ')}`;

      // 步骤2：调用Agent B（LLM）做智能化合规校验
      const llmPrompt = this.buildAgentBPrompt(apiDoc, agentResponse, basicCheckMsg);
      const llmResponse = await this.callLLM(llmPrompt);
      
      // 步骤3：解析LLM返回结果（Agent B的智能决策）
      const llmResult = JSON.parse(llmResponse);
      
      // 步骤4：生成合规凭证（自动化，硬编码）
      const credential = this.generateCredential();

      return {
        pass: basicCheckPass && llmResult.pass,
        checkDetails: `${basicCheckMsg}；${llmResult.checkDetails}`,
        violationRules: llmResult.violationRules,
        correctionSuggestion: llmResult.correctionSuggestion,
        credential
      };
    } catch (error) {
      throw new Error(`合规校验服务异常：${(error as Error).message}`);
    }
  }

  /**
   * 构建Agent B的Prompt（核心：贴合技术总监身份，聚焦合规校验）
   * 特点：小而专，仅包含当前校验所需信息，无冗余内容
   */
  private buildAgentBPrompt(
    apiDoc: MCPAPIDoc,
    agentResponse: AgentResponse,
    basicCheckMsg: string
  ): string {
    return `
      你是保险事业部技术总监（Agent B），负责MCP合规校验的最终决策。
      请严格参考以下MCP接口文档，对执行Agent返回的保险领域内容做合规校验：
      
      ### MCP接口文档（仅参考当前接口）
      接口名称：${apiDoc.apiName}
      合规核心规则：${apiDoc.complianceRules.join('；')}
      字段格式规则：${JSON.stringify(apiDoc.fieldFormatRules)}
      
      ### 执行Agent返回内容
      保险领域场景：${agentResponse.domainScene}
      MCP能力类型：${agentResponse.capabilityType}
      核心返回内容：${JSON.stringify(agentResponse.content)}
      
      ### 基础校验结果（必填字段）
      ${basicCheckMsg}
      
      ### 你的校验要求（技术总监决策权）
      1. 基于基础校验结果，进一步校验是否符合所有合规规则；
      2. 若违规，明确指出违反的规则，并给出保险领域专属的整改建议；
      3. 仅返回JSON格式，包含字段：pass（布尔）、checkDetails（校验详情）、violationRules（违规规则数组，可选）、correctionSuggestion（整改建议）；
      4. 整改建议需贴合保险业务实际，具备可落地性（如"标题移除'百分百赔付'敏感词，替换为'赔付比例以条款为准'"）。
    `.trim();
  }

  /**
   * 调用LLM（Agent B的核心能力载体）
   * 仅封装HTTP调用，无业务逻辑硬编码
   */
  private async callLLM(prompt: string): Promise<string> {
    const response = await axios.post(
      this.llmConfig.apiUrl,
      {
        model: this.llmConfig.model,
        messages: [{ role: 'user', content: prompt }],
        temperature: 0.1 // 低随机性，保证校验严谨性
      },
      {
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${this.llmConfig.apiKey}`
        }
      }
    );
    return response.data.choices[0].message.content;
  }

  /**
   * 基础必填字段校验（仅有的业务硬编码，兜底用）
   */
  private checkRequiredFields(apiDoc: MCPAPIDoc, content: Record<string, any>): boolean {
    return apiDoc.requiredFields.every(field => content[field] !== undefined && content[field] !== '');
  }

  /**
   * 获取缺失的必填字段
   */
  private getMissingFields(apiDoc: MCPAPIDoc, content: Record<string, any>): string[] {
    return apiDoc.requiredFields.filter(field => content[field] === undefined || content[field] === '');
  }

  /**
   * 生成合规凭证编号（自动化，硬编码）
   * 格式：COMP-{时间戳}-{随机数}
   */
  private generateCredential(): string {
    const timestamp = Date.now();
    const random = Math.floor(Math.random() * 10000);
    return `COMP-${timestamp}-${random.toString().padStart(4, '0')}`;
  }
}

import { ComplianceCheckService } from './ComplianceCheckService';
import { MCPAPIDoc, AgentResponse } from './types';

/**
 * 调用示例：模拟保险科普文发布的MCP合规校验
 * Agent B参考接口文档，校验insurance-d返回的创作内容

async function demo() {
  // 1. 初始化合规校验服务
  const checkService = new ComplianceCheckService();

  // 2. 模拟MCP接口文档（Agent B参考的依据，规则用自然语言描述）
  const apiDoc: MCPAPIDoc = {
    apiName: "保险科普文发布合规校验",
    complianceRules: [
      "禁止使用'百分百赔付'/'保证续保'等保险敏感词",
      "文章标题需包含【保险科普】前缀",
      "目标受众需明确（如30-40岁职场人群）",
      "内容不得夸大保险产品收益"
    ],
    requiredFields: ["title", "content", "targetAudience"],
    fieldFormatRules: {
      "title": "长度不超过30字",
      "content": "字数在800-2000字之间"
    },
    errorCodeMapping: {
      "FIELD_MISSING": "必填字段缺失",
      "RULE_VIOLATION": "违反合规规则"
    }
  };

  // 3. 模拟insurance-d返回的创作内容（含违规点）
  const agentResponse: AgentResponse = {
    content: {
      title: "百分百赔付的医疗险，你值得拥有", // 含敏感词
      content: "这款医疗险保证续保，收益远超存款！", // 违规+夸大
      targetAudience: "30-40岁" // 必填字段完整
    },
    domainScene: "医疗险科普-公众号发布",
    capabilityType: "wechat_publish"
  };

  // 4. Agent B执行合规校验
  const result = await checkService.checkCompliance(apiDoc, agentResponse);

  // 5. 输出校验结果（合规凭证）
  console.log("=== MCP合规校验结果 ===");
  console.log(`合规凭证：${result.credential}`);
  console.log(`是否通过：${result.pass}`);
  console.log(`校验详情：${result.checkDetails}`);
  console.log(`违规规则：${result.violationRules?.join('；') || '无'}`);
  console.log(`整改建议：${result.correctionSuggestion}`);
}

三、核心设计亮点（最大化智能化，弱化硬编码）
智能校验核心靠 LLM：除了 “必填字段是否存在” 的基础兜底，所有合规规则（敏感词、格式、业务约束）都由 Agent B（LLM）基于自然语言接口文档智能解析，无需硬编码if-else判断敏感词、标题格式等。
Prompt 小而专：Agent B 的 Prompt 仅包含当前校验所需的 “接口文档 + Agent 返回内容 + 基础校验结果”，无冗余信息，既降低 Prompt 长度，又保证 LLM 推理聚焦。
贴合业务身份：Prompt 中明确 Agent B 是 “保险事业部技术总监”，校验结果包含 “保险领域专属整改建议”，而非通用建议，符合领域超级智能体定位。
自动化凭证生成：合规凭证编号自动生成，作为校验结果的唯一标识，方便后续追溯，硬编码仅做格式拼接，无业务逻辑。
