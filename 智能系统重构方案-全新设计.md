## 一、核心设计理念

```

│                    全新智能化设计原则                              │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │

│     - 凡是能交给 LLM 决策的，绝不硬编码                           │
│     - 系统仅提供框架和约束，具体逻辑由 LLM 推理                   │
│     - 参考备份文件1：合规检查完全由 LLM 基于接口文档推理          │
│                                                                 │
│  2. 【标准化交互原则】                                            │
│     - Agent 之间交互必须遵循标准化格式                           │

│     - 参考备份文件2：通用交互框架定义                              │
│                                                                 │
│  3. 【可追溯原则】                                                │
│     - 每一次决策和执行都必须有完整记录                            │
│     - 支持历史回溯和问题诊断                                      │
│                                                                 │
│  4. 【容错原则】                                                  │
│     - 当 LLM 决策失败时，系统有降级方案                            │
│     - 从"下游补救"转向"上游约束"                                  │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### 1.2 与当前架构的核心差异

| 对比维度 | 当前架构 | 新设计方案 |
|---------|---------|-----------|
| **能力边界判定** | 执行 Agent 先做能力判定，再决定是否执行 | 执行 Agent 直接执行，返回标准格式结果 |
| **硬编码规则** | 较多硬编码判断逻辑（如三层检查） | 极少硬编码，仅做基础格式校验 |
| **Agent B 决策** | 基于固定流程和硬编码规则 | 完全基于 LLM 智能推理 |
| **容错机制** | 下游补救（发现问题再处理） | 上游约束（在源头就规范返回格式） |
| **扩展性** | 新增场景需修改代码 | 新增场景只需调整提示词 |

### 3.1 执行 Agent 标准返回（ExecutorStandardResult）

```typescript
/**
 * 执行 Agent 标准返回格式
 * 
 * 设计原则：
 * 1. 固定格式，便于验证
 * 2. 包含足够信息供 Agent B 决策
 * 3. 不做业务逻辑判断，仅描述执行情况

interface ExecutorStandardResult {
  // 核心状态
  status: 'COMPLETED' | 'PARTIAL' | 'BLOCKED' | 'FAILED';
  
  // 执行结果（任意格式，由 Agent B 解析）
  result?: any;

    description: string;
    context?: Record&lt;string, any&gt;;
  };
  
  // 已完成的工作
  completedWork?: {
    description: string;
    output?: any;
  }[];
  
  // 建议的下一步（可选，供 Agent B 参考）
  suggestedNext?: {
    type: 'USE_MCP' | 'ASK_USER' | 'RETRY' | 'OTHER';
    description: string;
    mcpSuggestion?: {
      capabilityType?: string;
      params?: Record&lt;string, any&gt;;
    };
    userSuggestion?: {
      infoNeeded: string;
      question: string;
    };
  };
  
  // 元数据
  metadata: {
    executionTime: number;
    timestamp: string;
    agentVersion: string;
  };
}
```

```typescript
/**
```

```typescript
 * 
 * 关键变化：
```

### 4.3 格式验证器（上游约束！）

```typescript
/**
 * 验证执行 Agent 返回格式
 * 
 * 关键设计：
 * 1. 仅验证格式，不验证内容
 * 2. 内容完全交给 Agent B（LLM）判断
 * 3. 从源头就规范返回格式，避免下游补救
 */
function validateExecutorResult(result: any): {
  isValid: boolean;
  errors: string[];
} {
  const errors: string[] = [];
  
  // 1. 检查核心字段
  if (!result.status) {
    errors.push('缺少 status 字段');
  } else if (!['COMPLETED', 'PARTIAL', 'BLOCKED', 'FAILED'].includes(result.status)) {
    errors.push('status 必须是 COMPLETED、PARTIAL、BLOCKED、FAILED 之一');
  }
  
  // 2. 检查 metadata
  if (!result.metadata) {
    errors.push('缺少 metadata 字段');
  } else {
    if (!result.metadata.executionTime) {
      errors.push('metadata 缺少 executionTime 字段');
    }
    if (!result.metadata.timestamp) {
      errors.push('metadata 缺少 timestamp 字段');
    }
  }
  
  // 3. 如果 status 不是 COMPLETED，检查 problem 字段
  if (result.status !== 'COMPLETED' &amp;&amp; !result.problem) {
    errors.push('非 COMPLETED 状态时必须提供 problem 字段');
  }
  
  // 4. 如果有 problem，检查 problem 格式
  if (result.problem) {
    if (!result.problem.type) {
      errors.push('problem 缺少 type 字段');
    }
    if (!result.problem.description) {
      errors.push('problem 缺少 description 字段');
    }
  }
  
  return {
    isValid: errors.length === 0,
    errors
  };
}

/**
 * 验证 Agent B 决策格式
 */
function validateAgentBDecision(decision: any): {
  isValid: boolean;
  errors: string[];
} {
  const errors: string[] = [];
  
  // 1. 检查决策类型
  if (!decision.decision) {
    errors.push('缺少 decision 字段');
  } else if (!['EXECUTE_MCP', 'ASK_USER', 'ACCEPT_RESULT', 'REJECT_AND_RETRY', 'MARK_FAILED'].includes(decision.decision)) {
    errors.push('decision 必须是 EXECUTE_MCP、ASK_USER、ACCEPT_RESULT、REJECT_AND_RETRY、MARK_FAILED 之一');
  }
  
  // 2. 检查推理过程
  if (!decision.reasoning) {
    errors.push('缺少 reasoning 字段');
  }
  
  // 3. 检查决策数据
  if (!decision.data) {
    errors.push('缺少 data 字段');
  }
  
  // 4. 根据决策类型检查对应字段
  if (decision.decision === 'EXECUTE_MCP' &amp;&amp; !decision.data.mcp) {
    errors.push('EXECUTE_MCP 决策时必须提供 data.mcp 字段');
  }
  
  if (decision.decision === 'ASK_USER' &amp;&amp; !decision.data.user) {
    errors.push('ASK_USER 决策时必须提供 data.user 字段');
  }
  
  // ... 其他类型检查
  
  return {
    isValid: errors.length === 0,
    errors
  };
}
```

### 4.4 Agent B 智能决策（完全 LLM 驱动！）

```typescript
 * Agent B 智能决策

 * 关键设计：
 * 1. 完全基于 LLM 推理
 * 2. 提供完整的上下文信息
 * 3. 返回标准化决策格式

async agentBIntelligentDecision(
  task: Task,
  executorResult: ExecutorStandardResult
): Promise&lt;void&gt; {
  console.log('[Agent B] 开始智能决策');
  
  // 1. 构建完整的执行上下文
  const context = await buildExecutionContext(task, executorResult);
  
  // 2. 构建提示词（完全开放给 LLM）
  const prompt = buildAgentBPrompt(context);
  
  // 3. 调用 LLM（单次调用，完全信任 LLM）
  const llmResponse = await callLLM('Agent B', '智能决策', '你是 Agent B，技术专家', prompt);
  
  // 4. 解析决策
  let decision: AgentBDecisionStandard;
  try {
    decision = JSON.parse(llmResponse);
  } catch (error) {
    console.error('[Agent B] 决策解析失败:', error);
    // 降级方案：标记为需要人工处理
    await markTaskAsNeedUser(task, 'Agent B 决策解析失败');
    return;
  }
  
  // 5. 验证决策格式
  const validationResult = validateAgentBDecision(decision);
  if (!validationResult.isValid) {
    console.error('[Agent B] 决策格式验证失败:', validationResult.errors);
    // 降级方案：标记为需要人工处理
    await markTaskAsNeedUser(task, `Agent B 决策格式错误: ${validationResult.errors.join(', ')}`);
    return;
  }
  
  // 6. 执行决策
  await executeAgentBDecision(task, decision);
}
```
