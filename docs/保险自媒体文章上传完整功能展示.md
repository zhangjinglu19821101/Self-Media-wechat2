```typescript
/**
 * 微信公众号 MCP 工具封装
 */

import {
  getAccessToken,
  addDraft,
  getDraftList,
  deleteDraft,
  uploadMedia,
} from '@/lib/wechat-official-account/api';

// ============================================
// 接口 1：上传素材（对应 capability.actionName = 'wechatUploadMedia'）
// ============================================
export interface WechatUploadMediaParams {
  accountId: string;
  mediaType: 'image';
  fileUrl?: string;
  fileBase64?: string;
}

export async function wechatUploadMedia(
  params: WechatUploadMediaParams
): Promise<WechatMCPResult<{ media_id: string; url: string }>> {
  try {
    const { accountId, mediaType, fileUrl, fileBase64 } = params;

    // 验证参数
    if (!accountId) {
      return {
        success: false,
        error: '缺少 accountId 参数',
        metadata: { timestamp: Date.now() },
      };
    }

    if (!mediaType) {
      return {
        success: false,
        error: '缺少 mediaType 参数',
        metadata: { timestamp: Date.now() },
      };
    }

    if (!fileUrl && !fileBase64) {
      return {
        success: false,
        error: 'fileUrl 和 fileBase64 必须提供一个',
        metadata: { timestamp: Date.now() },
      };
    }

    // 获取公众号账号
    const account = getAccountById(accountId);
    if (!account) {
      return {
        success: false,
        error: `未找到账号: ${accountId}`,
        metadata: { timestamp: Date.now() },
      };
    }

    // 调用 API 上传素材
    const result = await uploadMedia(account, mediaType, {
      fileUrl,
      fileBase64,
    });

    return {
      success: true,
      data: result,
      metadata: {
        accountId: account.id,
        accountName: account.name,
        timestamp: Date.now(),
      },
    };
  } catch (error: any) {
    return {
      success: false,
      error: `上传素材失败: ${error.message}`,
      metadata: { timestamp: Date.now() },
    };
  }
}

// ============================================
// 接口 2：添加草稿（对应 capability.actionName = 'wechatAddDraft'）
// ============================================
export interface WechatAddDraftParams {
  accountId: string;
  articles: WechatDraft[];
}

export async function wechatAddDraft(
  params: WechatAddDraftParams
): Promise<WechatMCPResult<{ media_id: string; create_time: number }>> {
  try {
    const { accountId, articles } = params;

    // 验证参数
    if (!accountId) {
      return {
        success: false,
        error: '缺少 accountId 参数',
        metadata: { timestamp: Date.now() },
      };
    }

    if (!articles || !Array.isArray(articles) || articles.length === 0) {
      return {
        success: false,
        error: '缺少 articles 参数或格式错误',
        metadata: { timestamp: Date.now() },
      };
    }

    // 获取公众号账号
    const account = getAccountById(accountId);
    if (!account) {
      return {
        success: false,
        error: `未找到账号: ${accountId}`,
        metadata: { timestamp: Date.now() },
      };
    }

    // 调用 API 添加草稿
    const result = await addDraft(account, articles);

    return {
      success: true,
      data: result,
      metadata: {
        accountId: account.id,
        accountName: account.name,
        timestamp: Date.now(),
      },
    };
  } catch (error: any) {
    return {
      success: false,
      error: `添加草稿失败: ${error.message}`,
      metadata: { timestamp: Date.now() },
    };
  }
}
```

### 5.2 轮询执行器代码

```typescript
/**
 * Agent B 轮询执行器
 * 定期轮询 insurance-d 的待执行任务
 */

import { db } from '@/lib/db';
import { agentSubTasks } from '@/lib/db/schema';
import { eq, and, lt } from 'drizzle-orm';
import { InsuranceMediaUploadOrchestrator } from './insurance-media-upload-orchestrator';

export class AgentBPoller {
  private isRunning = false;
  private pollInterval = 10000; // 10 秒

  /**
   * 启动轮询
   */
  async start() {
    if (this.isRunning) {
      console.log('[AgentBPoller] 已在运行中');
      return;
    }

    this.isRunning = true;
    console.log('[AgentBPoller] 启动轮询，间隔:', this.pollInterval, 'ms');

    // 立即执行一次
    await this.poll();

    // 定期轮询
    setInterval(() => {
      this.poll().catch(error => {
        console.error('[AgentBPoller] 轮询出错:', error);
      });
    }, this.pollInterval);
  }

  /**
   * 停止轮询
   */
  stop() {
    this.isRunning = false;
    console.log('[AgentBPoller] 停止轮询');
  }

  /**
   * 执行一次轮询
   */
  private async poll() {
    if (!this.isRunning) return;

    console.log('[AgentBPoller] 开始轮询...');

    try {
      // Step 1: 查询待执行任务
      const pendingTasks = await db
        .select()
        .from(agentSubTasks)
        .where(
          and(
            eq(agentSubTasks.status, 'in_progress'),
            eq(agentSubTasks.agentId, 'insurance-d')
          )
        );

      if (pendingTasks.length === 0) {
        console.log('[AgentBPoller] 没有待执行任务');
        return;
      }

      console.log(`[AgentBPoller] 找到 ${pendingTasks.length} 个待执行任务`);

      // Step 2: 逐个执行任务
      for (const task of pendingTasks) {
        await this.executeTask(task);
      }

    } catch (error) {
      console.error('[AgentBPoller] 轮询失败:', error);
    }
  }

  /**
   * 执行单个任务
   */
  private async executeTask(task: any) {
    console.log('[AgentBPoller] 执行任务:', task.id);

    try {
      // Step 1: 解析任务内容
      const uploadContext = this.parseTaskContent(task);
      if (!uploadContext) {
        console.warn('[AgentBPoller] 无法解析任务内容，跳过:', task.id);
        return;
      }

      // Step 2: 执行上传流程
      const result = await InsuranceMediaUploadOrchestrator.executeUpload(uploadContext);

      // Step 3: 更新任务状态
      await this.updateTaskStatus(task.id, result);

      // Step 4: 记录执行历史
      await this.recordExecutionHistory(task.id, result);

      console.log('[AgentBPoller] 任务执行完成:', task.id, result.success ? '成功' : '失败');

    } catch (error) {
      console.error('[AgentBPoller] 任务执行失败:', task.id, error);
      
      // 记录失败
      await this.updateTaskStatus(task.id, {
        success: false,
        error: error instanceof Error ? error.message : String(error)
      });
    }
  }

  /**
   * 解析任务内容
   */
  private parseTaskContent(task: any): any {
    // 从 taskDescription 或 metadata 中提取文章上传信息
    const taskDescription = task.taskDescription || '';
    const metadata = task.metadata || {};

    // 尝试解析 JSON
    try {
      if (metadata.uploadContext) {
        return metadata.uploadContext;
      }

      // 简单解析：假设 taskDescription 包含关键信息
      return {
        articleTitle: metadata.articleTitle || task.taskTitle,
        articleContent: metadata.articleContent || '',
        coverImageUrl: metadata.coverImageUrl || '',
        articleAuthor: metadata.articleAuthor,
        articleDigest: metadata.articleDigest
      };
    } catch {
      return null;
    }
  }

  /**
   * 更新任务状态
   */
  private async updateTaskStatus(taskId: string, result: any) {
    await db
      .update(agentSubTasks)
      .set({
        status: result.success ? 'completed' : 'failed',
        result: result,
        updatedAt: new Date()
      })
      .where(eq(agentSubTasks.id, taskId));
  }

  /**
   * 记录执行历史
   */
  private async recordExecutionHistory(taskId: string, result: any) {
    // 记录到 agent_sub_tasks_step_history 表
    // (实现省略)
  }
}
```

5. ✅ **完整上传流程**：两步执行（先上传素材，再上传文章）
6. ✅ **任务状态更新**：执行完成后更新 agent_sub_tasks 表

### 关键关联点

**核心问题：怎么通过 capability 中的数据体现出对应的调用方法？**

**答案：**
1. **toolName 字段**：指定使用哪个 MCP 工具（如 `"wechat"` → `TOOL_MAP["wechat"]`）
2. **actionName 字段**：指定调用工具的哪个方法（如 `"wechatUploadMedia"` → `WechatMCPTools.wechatUploadMedia()`）
3. **paramExamples / paramTemplate / interfaceSchema**：告诉 Agent B 如何生成参数
4. **agent_response_spec**：告诉 Agent B 如何返回结果

这样，Agent B 只需要读取 capability_list 表的记录，就能知道：
- 用什么工具（toolName）
- 调什么方法（actionName）
- 传什么参数（paramExamples / paramTemplate / interfaceSchema）
- 怎么返回（agent_response_spec）
