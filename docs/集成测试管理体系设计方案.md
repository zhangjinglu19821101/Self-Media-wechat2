// src/app/api/test/execute/route.ts

import { NextRequest, NextResponse } from 'next/server';
import testRegistry from '@/lib/test/test-cases';
import { db } from '@/lib/db';
import { testExecutions } from '@/lib/db/schema';

/**
 * 统一测试执行入口
 * 
 * GET  /api/test/execute?action=list               - 列出所有用例
 * GET  /api/test/execute?action=suites             - 列出所有套件
 * POST /api/test/execute?action=run                - 执行测试
 * GET  /api/test/execute?action=status&id=xxx      - 查询状态
 * GET  /api/test/execute?action=report&id=xxx      - 获取报告
 * GET  /api/test/execute?action=history            - 查询历史
 */
export async function GET(request: NextRequest) {
  const searchParams = request.nextUrl.searchParams;
  const action = searchParams.get('action');

  try {
    switch (action) {
      case 'list':
        return listTestCases(searchParams);
      case 'suites':
        return listTestSuites();
      case 'status':
        return getExecutionStatus(searchParams);
      case 'report':
        return getTestReport(searchParams);
      case 'history':
        return getExecutionHistory(searchParams);
      default:
        return NextResponse.json({
          success: false,
          error: '未知action，支持: list, suites, status, report, history'
        }, { status: 400 });
    }
  } catch (error) {
    console.error('[TestExecute] GET error:', error);
    return NextResponse.json({
      success: false,
      error: error instanceof Error ? error.message : '执行失败'
    }, { status: 500 });
  }
}

export async function POST(request: NextRequest) {
  const searchParams = request.nextUrl.searchParams;
  const action = searchParams.get('action');

  try {
    switch (action) {
      case 'run':
        return runTests(await request.json());
      case 'cleanup':
        return cleanupTestData();
      default:
        return NextResponse.json({
          success: false,
          error: '未知action，支持: run, cleanup'
        }, { status: 400 });
    }
  } catch (error) {
    console.error('[TestExecute] POST error:', error);
    return NextResponse.json({
      success: false,
      error: error instanceof Error ? error.message : '执行失败'
    }, { status: 500 });
  }
}

// ========== 具体实现 ==========

async function listTestCases(searchParams: URLSearchParams) {
  const category = searchParams.get('category');
  const tag = searchParams.get('tag');
  const priority = searchParams.get('priority');

  let testCases = testRegistry.getAll();

  if (category) {
    testCases = testCases.filter(tc => tc.category === category);
  }
  if (tag) {
    testCases = testCases.filter(tc => tc.tags.includes(tag));
  }
  if (priority) {
    testCases = testCases.filter(tc => tc.priority === priority);
  }

  return NextResponse.json({
    success: true,
    data: {
      total: testCases.length,
      testCases: testCases.map(tc => ({
        id: tc.id,
        name: tc.name,
        description: tc.description,
        category: tc.category,
        priority: tc.priority,
        tags: tc.tags
      }))
    }
  });
}

async function listTestSuites() {
  const suites = testRegistry.getAllSuites();

  return NextResponse.json({
    success: true,
    data: {
      total: suites.length,
      suites: suites.map(suite => ({
        id: suite.id,
        name: suite.name,
        description: suite.description,
        testCaseCount: suite.testCases.length,
        parallel: suite.parallel
      }))
    }
  });
}

async function runTests(params: {
  testId?: string;
  suiteId?: string;
  category?: string;
  tags?: string[];
  parallel?: boolean;
}) {
  const { testId, suiteId, category, tags, parallel } = params;

  // 1. 确定要执行的用例
  let testCases: TestCase[] = [];

  if (testId) {
    const tc = testRegistry.getTestCase(testId);
    if (!tc) {
      return NextResponse.json({
        success: false,
        error: `测试用例 ${testId} 不存在`
      }, { status: 404 });
    }
    testCases = [tc];
  } else if (suiteId) {
    const suite = testRegistry.getSuite(suiteId);
    if (!suite) {
      return NextResponse.json({
        success: false,
        error: `测试套件 ${suiteId} 不存在`
      }, { status: 404 });
    }
    testCases = suite.testCases.map(id => testRegistry.getTestCase(id)).filter(Boolean);
  } else if (category) {
    testCases = testRegistry.getByCategory(category as TestCategory);
  } else if (tags && tags.length > 0) {
    testCases = testRegistry.getByTag(tags[0]);
  } else {
    return NextResponse.json({
      success: false,
      error: '请指定 testId, suiteId, category 或 tags'
    }, { status: 400 });
  }

  // 2. 创建执行记录
  const executionId = generateExecutionId();
  await db.insert(testExecutions).values({
    id: executionId,
    testCases: testCases.map(tc => tc.id),
    status: 'running',
    startTime: new Date(),
    params
  });

  // 3. 异步执行测试
  executeTestsAsync(executionId, testCases, parallel || false);

  return NextResponse.json({
    success: true,
    data: {
      executionId,
      testCount: testCases.length,
      status: 'running',
      message: '测试已开始执行'
    }
  });
}

async function executeTestsAsync(
  executionId: string,
  testCases: TestCase[],
  parallel: boolean
) {
  const results = [];

  if (parallel) {
    // 并行执行
    const promises = testCases.map(tc => executeSingleTest(tc));
    const parallelResults = await Promise.allSettled(promises);
    results.push(...parallelResults);
  } else {
    // 串行执行
    for (const testCase of testCases) {
      const result = await executeSingleTest(testCase);
      results.push(result);
    }
  }

  // 更新执行记录
  await db.update(testExecutions)
    .set({
      status: 'completed',
      endTime: new Date(),
      results
    })
    .where(eq(testExecutions.id, executionId));
}

async function executeSingleTest(testCase: TestCase): Promise<TestResult> {
  const startTime = Date.now();
  
  try {
    const response = await fetch(`http://localhost:5000${testCase.entryPoint}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(testCase.params || {})
    });

    const data = await response.json();
    const duration = Date.now() - startTime;

    return {
      testId: testCase.id,
      name: testCase.name,
      status: data.success ? 'passed' : 'failed',
      duration,
      message: data.message || data.error,
      data: data.data
    };
  } catch (error) {
    return {
      testId: testCase.id,
      name: testCase.name,
      status: 'failed',
      duration: Date.now() - startTime,
      message: error instanceof Error ? error.message : '执行异常',
      error: String(error)
    };
  }
}
