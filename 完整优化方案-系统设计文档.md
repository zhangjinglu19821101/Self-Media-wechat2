```typescript
type AgentBReviewDecision =

  | 'NEED_USER'         // 需要用户确认

  taskId: string;                      // 任务ID
  reviewType: 'pre_completed' | 'pre_need_support';  // 评审类型

  confidence: number;                   // 置信度（0-1）
  
  // 决策数据（根据decision类型不同而不同）
  data?: {
    // NEED_REVISE: 修改意见
```

```typescript
/**
 * 功能开关配置
 */
interface FeatureFlags {
  // 总开关
  enableAgentBReview: boolean;
  
  // 细粒度开关
  enablePreCompletedReview: boolean;
  enablePreNeedSupportReview: boolean;
  
  // 白名单机制
  enabledTaskIds?: Set<string>;
  enabledExecutors?: Set<string>;
  
  // 灰度发布
  rolloutPercentage?: number;  // 0-100
  
  // 降级开关
  enableFallback: boolean;
}

/**
 * 功能开关管理器
 */
class FeatureFlagManager {
  private config: FeatureFlags;
  
  constructor() {
    this.loadConfig();
  }
  
  private loadConfig() {
    this.config = {
      enableAgentBReview: process.env.ENABLE_AGENT_B_REVIEW === 'true',
      enablePreCompletedReview: process.env.ENABLE_PRE_COMPLETED_REVIEW === 'true',
      enablePreNeedSupportReview: process.env.ENABLE_PRE_NEED_SUPPORT_REVIEW === 'true',
      enabledTaskIds: new Set(process.env.ENABLED_TASK_IDS?.split(',') || []),
      enabledExecutors: new Set(process.env.ENABLED_EXECUTORS?.split(',') || []),
      rolloutPercentage: parseInt(process.env.ROLLOUT_PERCENTAGE || '0'),
      enableFallback: true,
    };
  }
  
  /**
   * 检查是否启用 Agent B 评审
   */
  isEnabled(task: typeof agentSubTasks.$inferSelect): boolean {
    // 总开关
    if (!this.config.enableAgentBReview) {
      return false;
    }
    
    // 任务ID白名单
    if (this.config.enabledTaskIds?.size > 0) {
      return this.config.enabledTaskIds.has(task.id);
    }
    
    // 执行者白名单
    if (this.config.enabledExecutors?.size > 0) {
      return this.config.enabledExecutors.has(task.fromParentsExecutor);
    }
    
    // 灰度发布
    if (this.config.rolloutPercentage && this.config.rolloutPercentage < 100) {
      const hash = this.hashTaskId(task.id);
      return hash % 100 < this.config.rolloutPercentage;
    }
    
    return true;
  }
  
  /**
   * 检查是否启用 pre_completed 评审
   */
  isPreCompletedReviewEnabled(task: typeof agentSubTasks.$inferSelect): boolean {
    return this.isEnabled(task) && this.config.enablePreCompletedReview;
  }
  
  /**
   * 检查是否启用 pre_need_support 评审
   */
  isPreNeedSupportReviewEnabled(task: typeof agentSubTasks.$inferSelect): boolean {
    return this.isEnabled(task) && this.config.enablePreNeedSupportReview;
  }
  
  /**
   * 简单的任务ID哈希（用于灰度发布）
   */
  private hashTaskId(taskId: string): number {
    let hash = 0;
    for (let i = 0; i < taskId.length; i++) {
      const char = taskId.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash;
    }
    return Math.abs(hash);
  }
}
```

### 3. 状态机处理逻辑

```typescript
/**
 * 扩展 SubtaskExecutionEngine 类
 */
class SubtaskExecutionEngine {
  private featureFlagManager: FeatureFlagManager;
  private agentBReviewer: AgentBReviewer;
  
  constructor() {
    this.featureFlagManager = new FeatureFlagManager();
    this.agentBReviewer = new AgentBReviewer();
  }
  
  /**
   * 分支2：处理 pre_completed 状态
   */
  private async handlePreCompletedStatus(task: typeof agentSubTasks.$inferSelect) {
    console.log(`[状态机] 处理 pre_completed: ${task.id}`);
    
    // 检查是否启用 Agent B 评审
    if (!this.featureFlagManager.isPreCompletedReviewEnabled(task)) {
      console.log(`[状态机] Agent B 评审未启用，使用降级逻辑`);
      await this.handlePreCompletedFallback(task);
      return;
    }
    
    try {
      // 1. 调用 Agent B 进行评审
      console.log(`[状态机] 调用 Agent B 进行评审...`);
      const reviewResult = await this.agentBReviewer.reviewPreCompleted(task);
      
      // 2. 保存评审结果
      await this.saveReviewResult(task.id, reviewResult);
      
      // 3. 根据评审结果处理
      await this.processPreCompletedReviewResult(task, reviewResult);
      
    } catch (error) {
      console.error(`[状态机] Agent B 评审失败，使用降级逻辑:`, error);
      await this.handlePreCompletedFallback(task);
    }
  }
  
  /**
   * 处理 pre_completed 评审结果
   */
  private async processPreCompletedReviewResult(
    task: typeof agentSubTasks.$inferSelect,
    reviewResult: AgentBReviewResult
  ) {
    console.log(`[状态机] 处理评审结果: ${reviewResult.decision}`);
    
    switch (reviewResult.decision) {
      case 'APPROVE':
        // 批准：直接完成
        await this.markTaskCompleted(
          task,
          task.executionResult ? JSON.parse(task.executionResult) : { success: true }
        );
        console.log(`[状态机] Agent B 批准，标记为 completed`);
        break;
        
      case 'NEED_REVISE':
        // 需要修改：让执行Agent重新执行
        await this.retryTaskWithRevision(task, reviewResult);
        console.log(`[状态机] Agent B 要求修改，回到 pending 状态`);
        break;
        
      case 'NEED_USER':
        // 需要用户确认
        await this.markTaskWaitingUser(task, reviewResult);
        console.log(`[状态机] Agent B 需要用户确认，标记为 waiting_user`);
        break;
        
      default:
        console.error(`[状态机] 未知的决策类型: ${reviewResult.decision}`);
        // 未知决策时，使用降级逻辑
        await this.handlePreCompletedFallback(task);
    }
  }
  
  /**
   * 让执行Agent重新执行（带修改意见）
   */
  private async retryTaskWithRevision(
    task: typeof agentSubTasks.$inferSelect,
    reviewResult: AgentBReviewResult
  ) {
    // 检查重试次数
    const currentRetryCount = task.metadata?.retryCount || 0;
    const maxRetries = task.metadata?.maxRetries || 3;
    
    if (currentRetryCount >= maxRetries) {
      console.error(`[状态机] 达到最大重试次数 ${maxRetries}，标记为失败`);
      await this.markTaskFailed(task, '达到最大重试次数');
      return;
    }
    
    // 更新任务状态为 pending，保存修改意见
    await db
      .update(agentSubTasks)
      .set({
        status: 'pending',
        metadata: {
          ...task.metadata,
          retryCount: currentRetryCount + 1,
          lastRetryReason: reviewResult.reasoning,
          retryHistory: [
            ...(task.metadata?.retryHistory || []),
            {
              retryId: `retry-${Date.now()}`,
              retryAt: new Date(),
              reason: reviewResult.reasoning,
              reviewResult,
            },
          ],
          agentBRevision: reviewResult.data?.revisionInstructions,
        },
        updatedAt: getCurrentBeijingTime(),
      })
      .where(eq(agentSubTasks.id, task.id));
  }
  
  /**
   * 标记任务为 waiting_user
   */
  private async markTaskWaitingUser(
    task: typeof agentSubTasks.$inferSelect,
    reviewResult: AgentBReviewResult
  ) {
    await db
      .update(agentSubTasks)
      .set({
        status: 'waiting_user',
        metadata: {
          ...task.metadata,
          currentReview: reviewResult,
        },
        updatedAt: getCurrentBeijingTime(),
      })
      .where(eq(agentSubTasks.id, task.id));
  }
  
  /**
   * pre_completed 降级逻辑（原有逻辑）
   */
  private async handlePreCompletedFallback(task: typeof agentSubTasks.$inferSelect) {
    console.log(`[状态机] 使用降级逻辑：直接标记为 completed`);
    await this.markTaskCompleted(
      task,
      task.executionResult ? JSON.parse(task.executionResult) : { success: true }
    );
  }
  
  /**
   * 分支3：处理 pre_need_support 状态
   */
  private async handlePreNeedSupportStatus(task: typeof agentSubTasks.$inferSelect) {
    console.log(`[状态机] 处理 pre_need_support: ${task.id}`);
    
    // 检查是否启用 Agent B 评审
    if (!this.featureFlagManager.isPreNeedSupportReviewEnabled(task)) {
      console.log(`[状态机] Agent B 评审未启用，使用降级逻辑`);
      await this.handlePreNeedSupportFallback(task);
      return;
    }
    
    try {
      // 1. 调用 Agent B 进行评审
      console.log(`[状态机] 调用 Agent B 进行评审...`);
      const reviewResult = await this.agentBReviewer.reviewPreNeedSupport(task);
      
      // 2. 保存评审结果
      await this.saveReviewResult(task.id, reviewResult);
      
      // 3. 根据评审结果处理
      await this.processPreNeedSupportReviewResult(task, reviewResult);
      
    } catch (error) {
      console.error(`[状态机] Agent B 评审失败，使用降级逻辑:`, error);
      await this.handlePreNeedSupportFallback(task);
    }
  }
  
  /**
   * 处理 pre_need_support 评审结果
   */
  private async processPreNeedSupportReviewResult(
    task: typeof agentSubTasks.$inferSelect,
    reviewResult: AgentBReviewResult
  ) {
    console.log(`[状态机] 处理评审结果: ${reviewResult.decision}`);
    
    switch (reviewResult.decision) {
      case 'CAN_HELP':
        // Agent B 可以提供帮助：让执行Agent重新执行
        await this.retryTaskWithHelp(task, reviewResult);
        console.log(`[状态机] Agent B 提供帮助，回到 pending 状态`);
        break;
        
      case 'NEED_USER':
        // 需要用户确认
        await this.markTaskWaitingUser(task, reviewResult);
        console.log(`[状态机] Agent B 需要用户确认，标记为 waiting_user`);
        break;
        
      case 'CANNOT_HELP':
        // Agent B 也帮不了：标记失败
        await this.markTaskFailed(task, reviewResult.reasoning);
        console.log(`[状态机] Agent B 也帮不了，标记为 failed`);
        break;
        
      default:
        console.error(`[状态机] 未知的决策类型: ${reviewResult.decision}`);
        // 未知决策时，使用降级逻辑
        await this.handlePreNeedSupportFallback(task);
    }
  }
  
  /**
   * 让执行Agent重新执行（带Agent B的帮助）
   */
  private async retryTaskWithHelp(
    task: typeof agentSubTasks.$inferSelect,
    reviewResult: AgentBReviewResult
  ) {
    // 检查重试次数
    const currentRetryCount = task.metadata?.retryCount || 0;
    const maxRetries = task.metadata?.maxRetries || 3;
    
    if (currentRetryCount >= maxRetries) {
      console.error(`[状态机] 达到最大重试次数 ${maxRetries}，标记为失败`);
      await this.markTaskFailed(task, '达到最大重试次数');
      return;
    }
    
    // 更新任务状态为 pending，保存 Agent B 的帮助信息
    await db
      .update(agentSubTasks)
      .set({
        status: 'pending',
        metadata: {
          ...task.metadata,
          retryCount: currentRetryCount + 1,
          lastRetryReason: reviewResult.reasoning,
          retryHistory: [
            ...(task.metadata?.retryHistory || []),
            {
              retryId: `retry-${Date.now()}`,
              retryAt: new Date(),
              reason: reviewResult.reasoning,
              reviewResult,
            },
          ],
          agentBHelp: reviewResult.data?.helpInfo,
        },
        updatedAt: getCurrentBeijingTime(),
      })
      .where(eq(agentSubTasks.id, task.id));
  }
  
  /**
   * pre_need_support 降级逻辑（原有逻辑）
   */
  private async handlePreNeedSupportFallback(task: typeof agentSubTasks.$inferSelect) {
    console.log(`[状态机] 使用降级逻辑：直接标记为 waiting_user`);
    await db
      .update(agentSubTasks)
      .set({
        status: 'waiting_user',
        updatedAt: getCurrentBeijingTime(),
      })
      .where(eq(agentSubTasks.id, task.id));
  }
  
  /**
   * 保存评审结果
   */
  private async saveReviewResult(taskId: string, reviewResult: AgentBReviewResult) {
```
