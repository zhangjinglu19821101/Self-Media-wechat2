## 🤔 问题1：什么是"失败时自动降级到原有逻辑"？

### 解释

**降级逻辑 = 保险机制**

当新逻辑（Agent B 评审）出现问题时，自动回退到原来的逻辑。

### 原来的逻辑（降级方案）

```typescript
// handlePreCompletedStatus 的降级逻辑
private async handlePreCompletedFallback(task: typeof agentSubTasks.$inferSelect) {
  // 就是原来的逻辑：直接标记为 completed
  await this.markTaskCompleted(
    task,
    task.executionResult ? JSON.parse(task.executionResult) : { success: true }
  );
}

// handlePreNeedSupportStatus 的降级逻辑
private async handlePreNeedSupportFallback(task: typeof agentSubTasks.$inferSelect) {
  // 就是原来的逻辑：直接标记为 waiting_user
  await db
    .update(agentSubTasks)
    .set({
      status: 'waiting_user',
      updatedAt: getCurrentBeijingTime(),
    })
    .where(eq(agentSubTasks.id, task.id));
}
```

### 什么时候触发降级？

```typescript
try {
  // 1. 调用 Agent B 进行评审（新逻辑）
  const reviewResult = await this.agentBReviewer.reviewPreCompleted(task);
  
  // 2. 根据评审结果处理（新逻辑）
  await this.processPreCompletedReviewResult(task, reviewResult);
  
} catch (error) {
  // ❌ 新逻辑出错了！
  console.error(`[状态机] Agent B 评审失败，使用降级逻辑:`, error);
  
  // ✅ 自动降级：使用原来的逻辑
  await this.handlePreCompletedFallback(task);
}
```

### 为什么需要降级？

| 场景 | 说明 | 降级后的行为 |
|------|------|-------------|
| Agent B API 报错 | 调用 LLM 失败 | 直接标记为 completed / waiting_user |
| 评审响应解析失败 | JSON 解析出错 | 直接标记为 completed / waiting_user |
| 数据库操作失败 | 更新状态出错 | 记录错误，不阻塞流程 |
| 未知异常 | 其他任何错误 | 回退到原有逻辑 |

**总结：降级逻辑 = 安全网，确保新逻辑出错时不影响现有流程！**

```typescript
/**
 * Agent B 评审器（简化版）
 */
class AgentBReviewer {
  /**
   * 评审 pre_completed 状态的任务
   */
  async reviewPreCompleted(
    task: typeof agentSubTasks.$inferSelect
  ): Promise<AgentBReviewResult> {
    console.log(`[Agent B 评审器] 开始评审 pre_completed: ${task.id}`);
    
    try {
      // 1. 构建评审提示词
      const prompt = this.buildPreCompletedReviewPrompt(task);
      
      // 2. 调用 Agent B 进行评审
      const response = await callLLM(
        'B',
        '评审执行Agent的结果',
        loadAgentPrompt('B'),
        prompt
      );
      
      // 3. 解析评审结果
      const reviewResult = this.parseReviewResponse(response);
      console.log(`[Agent B 评审器] 评审完成:`, reviewResult);
      return reviewResult;
      
    } catch (error) {
      console.error(`[Agent B 评审器] 评审失败:`, error);
      // 评审失败时，保守批准
      return {
        decision: 'APPROVE',
        reasoning: 'Agent B 评审失败，保守批准',
      };
    }
  }
  
  /**
   * 评审 pre_need_support 状态的任务
   */
  async reviewPreNeedSupport(
    task: typeof agentSubTasks.$inferSelect
  ): Promise<AgentBReviewResult> {
    console.log(`[Agent B 评审器] 开始评审 pre_need_support: ${task.id}`);
    
    try {
      // 1. 构建评审提示词
      const prompt = this.buildPreNeedSupportReviewPrompt(task);
      
      // 2. 调用 Agent B 进行评审
      const response = await callLLM(
        'B',
        '评审执行Agent的求助',
        loadAgentPrompt('B'),
        prompt
      );
      
      // 3. 解析评审结果
      const reviewResult = this.parseReviewResponse(response, 'pre_need_support');
      console.log(`[Agent B 评审器] 评审完成:`, reviewResult);
      return reviewResult;
      
    } catch (error) {
      console.error(`[Agent B 评审器] 评审失败:`, error);
      // 评审失败时，转人工（不用 CANNOT_HELP）
      return {
        decision: 'NEED_USER',
        reasoning: 'Agent B 评审失败，转人工处理',
      };
    }
  }
  
  /**
   * 构建 pre_completed 评审提示词
   */
  private buildPreCompletedReviewPrompt(task: typeof agentSubTasks.$inferSelect): string {
    const executionResult = task.executionResult 
      ? JSON.parse(task.executionResult)
      : null;
    
    return `
【任务信息】
任务标题：${task.taskTitle}
任务描述：${task.taskDescription}
执行者：${task.fromParentsExecutor}

【执行Agent的结果】
${executionResult ? JSON.stringify(executionResult, null, 2) : '无'}

【你的目标】
作为 Agent B（集团管理者），评审执行Agent的工作结果。

【输出要求】
请用 JSON 格式输出：
{
  "decision": "APPROVE" | "NEED_REVISE" | "NEED_USER",
  "reasoning": "详细的推理过程",
  "data": {
    "revisionInstructions": "修改指令（NEED_REVISE时）",
    "pendingKeyFields": [...],  // NEED_USER时
    "promptMessage": {...}       // NEED_USER时
  }
}

【决策说明】
- APPROVE：结果合格，直接完成
- NEED_REVISE：需要修改，让执行Agent重新执行
- NEED_USER：需要用户确认
`;
  }
  
  /**
   * 构建 pre_need_support 评审提示词
   */
  private buildPreNeedSupportReviewPrompt(task: typeof agentSubTasks.$inferSelect): string {
    const executionResult = task.executionResult 
      ? JSON.parse(task.executionResult)
      : null;
    
    return `
【任务信息】
任务标题：${task.taskTitle}
任务描述：${task.taskDescription}
执行者：${task.fromParentsExecutor}

【执行Agent的求助】
${executionResult ? JSON.stringify(executionResult, null, 2) : '无'}

【你的目标】
作为 Agent B（集团管理者），判断能否帮助执行Agent，或者需要用户介入。

【输出要求】
请用 JSON 格式输出：
{
  "decision": "CAN_HELP" | "NEED_USER",
  "reasoning": "详细的推理过程",
  "data": {
    "helpInfo": {
      "helpType": "information",
      "content": "帮助内容"
    },
    "pendingKeyFields": [...],  // NEED_USER时
    "promptMessage": {...}       // NEED_USER时
  }
}

【决策说明】
- CAN_HELP：你可以提供帮助，让执行Agent重新执行
- NEED_USER：需要用户介入（你帮不了时也选这个）
`;
  }
  
  /**
   * 解析评审响应
   */
  private parseReviewResponse(
    response: string,
    reviewType: 'pre_completed' | 'pre_need_support'
  ): AgentBReviewResult {
    try {
      const jsonMatch = response.match(/\{[\s\S]*\}/);
      if (jsonMatch) {
        const parsed = JSON.parse(jsonMatch[0]);
        
        // 对于 pre_need_support，CANNOT_HELP 改为 NEED_USER
        let decision = parsed.decision;
        if (reviewType === 'pre_need_support' && decision === 'CANNOT_HELP') {
          console.log(`[Agent B 评审器] CANNOT_HELP 改为 NEED_USER`);
          decision = 'NEED_USER';
        }
        
        return {
          decision,
          reasoning: parsed.reasoning,
          data: parsed.data,
        };
      }
    } catch (e) {
      console.error('[Agent B 评审器] 解析响应失败:', e);
    }
    
    // 解析失败时的默认值
    return {
      decision: reviewType === 'pre_completed' ? 'APPROVE' : 'NEED_USER',
      reasoning: '解析评审响应失败，使用默认决策',
    };
  }
}
```

### 3. 状态机处理逻辑（简化版）

```typescript
/**
 * 扩展 SubtaskExecutionEngine 类（简化版）
 */
class SubtaskExecutionEngine {
  private agentBReviewer: AgentBReviewer;
  
  constructor() {
    this.agentBReviewer = new AgentBReviewer();
  }
  
  /**
   * 分支2：处理 pre_completed 状态
   */
  private async handlePreCompletedStatus(task: typeof agentSubTasks.$inferSelect) {
    console.log(`[状态机] 处理 pre_completed: ${task.id}`);
    
    try {
      // 1. 调用 Agent B 进行评审
      console.log(`[状态机] 调用 Agent B 进行评审...`);
      const reviewResult = await this.agentBReviewer.reviewPreCompleted(task);
      
      // 2. 根据评审结果处理
      await this.processPreCompletedReviewResult(task, reviewResult);
      
    } catch (error) {
      console.error(`[状态机] Agent B 评审失败，使用降级逻辑:`, error);
      await this.handlePreCompletedFallback(task);
    }
  }
  
  /**
   * 处理 pre_completed 评审结果
   */
  private async processPreCompletedReviewResult(
    task: typeof agentSubTasks.$inferSelect,
    reviewResult: AgentBReviewResult
  ) {
    console.log(`[状态机] 处理评审结果: ${reviewResult.decision}`);
    
    switch (reviewResult.decision) {
      case 'APPROVE':
        // 批准：直接完成
        await this.markTaskCompleted(
          task,
          task.executionResult ? JSON.parse(task.executionResult) : { success: true }
        );
        console.log(`[状态机] Agent B 批准，标记为 completed`);
        break;
        
      case 'NEED_REVISE':
        // 需要修改：让执行Agent重新执行
        await this.retryTaskWithRevision(task, reviewResult);
        console.log(`[状态机] Agent B 要求修改，回到 pending 状态`);
        break;
        
      case 'NEED_USER':
        // 需要用户确认
        await this.markTaskWaitingUser(task, reviewResult);
        console.log(`[状态机] Agent B 需要用户确认，标记为 waiting_user`);
        break;
        
      default:
        console.error(`[状态机] 未知的决策类型: ${reviewResult.decision}`);
        await this.handlePreCompletedFallback(task);
    }
  }
  
  /**
   * 让执行Agent重新执行（带修改意见）
   */
  private async retryTaskWithRevision(
    task: typeof agentSubTasks.$inferSelect,
    reviewResult: AgentBReviewResult
  ) {
    // 检查重试次数（最多3次）
    const currentRetryCount = task.metadata?.retryCount || 0;
    if (currentRetryCount >= 3) {
      console.error(`[状态机] 达到最大重试次数 3，标记为失败`);
      await this.markTaskFailed(task, '达到最大重试次数');
      return;
    }
    
    // 更新任务状态为 pending，保存修改意见
    await db
      .update(agentSubTasks)
      .set({
        status: 'pending',
        metadata: {
          ...task.metadata,
          retryCount: currentRetryCount + 1,
          lastRetryReason: reviewResult.reasoning,
          agentBRevision: reviewResult.data?.revisionInstructions,
        },
        updatedAt: getCurrentBeijingTime(),
      })
      .where(eq(agentSubTasks.id, task.id));
  }
  
  /**
   * 标记任务为 waiting_user
   */
  private async markTaskWaitingUser(
    task: typeof agentSubTasks.$inferSelect,
    reviewResult: AgentBReviewResult
  ) {
    await db
      .update(agentSubTasks)
      .set({
        status: 'waiting_user',
        metadata: {
          ...task.metadata,
          agentBReview: reviewResult,
        },
        updatedAt: getCurrentBeijingTime(),
      })
      .where(eq(agentSubTasks.id, task.id));
  }
  
  /**
   * pre_completed 降级逻辑（原有逻辑）
   */
  private async handlePreCompletedFallback(task: typeof agentSubTasks.$inferSelect) {
    console.log(`[状态机] 使用降级逻辑：直接标记为 completed`);
    await this.markTaskCompleted(
      task,
      task.executionResult ? JSON.parse(task.executionResult) : { success: true }
    );
  }
  
  /**
   * 分支3：处理 pre_need_support 状态
   */
  private async handlePreNeedSupportStatus(task: typeof agentSubTasks.$inferSelect) {
    console.log(`[状态机] 处理 pre_need_support: ${task.id}`);
    
    try {
      // 1. 调用 Agent B 进行评审
      console.log(`[状态机] 调用 Agent B 进行评审...`);
      const reviewResult = await this.agentBReviewer.reviewPreNeedSupport(task);
      
      // 2. 根据评审结果处理
      await this.processPreNeedSupportReviewResult(task, reviewResult);
      
    } catch (error) {
      console.error(`[状态机] Agent B 评审失败，使用降级逻辑:`, error);
      await this.handlePreNeedSupportFallback(task);
    }
  }
  
  /**
   * 处理 pre_need_support 评审结果
   */
  private async processPreNeedSupportReviewResult(
    task: typeof agentSubTasks.$inferSelect,
    reviewResult: AgentBReviewResult
  ) {
    console.log(`[状态机] 处理评审结果: ${reviewResult.decision}`);
    
    switch (reviewResult.decision) {
      case 'CAN_HELP':
        // Agent B 可以提供帮助：让执行Agent重新执行
        await this.retryTaskWithHelp(task, reviewResult);
        console.log(`[状态机] Agent B 提供帮助，回到 pending 状态`);
        break;
        
      case 'NEED_USER':
        // 需要用户确认
        await this.markTaskWaitingUser(task, reviewResult);
        console.log(`[状态机] Agent B 需要用户确认，标记为 waiting_user`);
        break;
        
      default:
        console.error(`[状态机] 未知的决策类型: ${reviewResult.decision}`);
        await this.handlePreNeedSupportFallback(task);
    }
  }
  
  /**
   * 让执行Agent重新执行（带Agent B的帮助）
   */
  private async retryTaskWithHelp(
    task: typeof agentSubTasks.$inferSelect,
    reviewResult: AgentBReviewResult
  ) {
    // 检查重试次数（最多3次）
    const currentRetryCount = task.metadata?.retryCount || 0;
    if (currentRetryCount >= 3) {
      console.error(`[状态机] 达到最大重试次数 3，标记为失败`);
      await this.markTaskFailed(task, '达到最大重试次数');
      return;
    }
    
    // 更新任务状态为 pending，保存 Agent B 的帮助信息
    await db
      .update(agentSubTasks)
      .set({
        status: 'pending',
        metadata: {
          ...task.metadata,
          retryCount: currentRetryCount + 1,
          lastRetryReason: reviewResult.reasoning,
          agentBHelp: reviewResult.data?.helpInfo,
        },
        updatedAt: getCurrentBeijingTime(),
      })
      .where(eq(agentSubTasks.id, task.id));
  }
  
  /**
   * pre_need_support 降级逻辑（原有逻辑）
   */
  private async handlePreNeedSupportFallback(task: typeof agentSubTasks.$inferSelect) {
    console.log(`[状态机] 使用降级逻辑：直接标记为 waiting_user`);
    await db
      .update(agentSubTasks)
      .set({
        status: 'waiting_user',
        updatedAt: getCurrentBeijingTime(),
      })
      .where(eq(agentSubTasks.id, task.id));
  }
}
```
