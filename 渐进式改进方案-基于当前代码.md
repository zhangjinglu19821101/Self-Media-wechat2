## 二、第二阶段：引入标准化格式（P1 - 本周内）

### 2.1 新增：执行 Agent 标准返回格式验证器

**目标**：在源头就确保执行 Agent 返回格式正确，避免下游补救。

**实现位置**：新建 `src/lib/validators/executor-result-validator.ts`

```typescript
/**
 * 执行 Agent 返回结果验证器
 * 
 * 设计原则：
 * 1. 仅验证格式，不验证内容
 * 2. 内容完全交给 Agent B（LLM）判断
 * 3. 从源头就规范返回格式
 */

interface ExecutorAgentResult {
  isNeedMcp: boolean;
  problem?: string;
  capabilityType?: string;
  executionResult?: any;
  isTaskDown: boolean;
}

export class ExecutorResultValidator {
  
  /**
   * 验证执行 Agent 返回结果格式
   */
  static validate(result: any): {
    isValid: boolean;
    errors: string[];
    normalizedResult?: ExecutorAgentResult;
  } {
    const errors: string[] = [];
    
    // 1. 检查核心字段
    if (typeof result.isNeedMcp !== 'boolean') {
      errors.push('isNeedMcp 必须是 boolean 类型');
    }
    
    if (typeof result.isTaskDown !== 'boolean') {
      errors.push('isTaskDown 必须是 boolean 类型');
    }
    
    // 2. 如果 isNeedMcp = true，检查 problem 字段
    if (result.isNeedMcp &amp;&amp; !result.problem) {
      errors.push('isNeedMcp = true 时必须提供 problem 字段');
    }
    
    // 3. 如果 isTaskDown = false 且 isNeedMcp = false，这是矛盾的
    if (!result.isNeedMcp &amp;&amp; !result.isTaskDown) {
      errors.push('isNeedMcp = false 且 isTaskDown = false 是矛盾的，至少有一个应该为 true');
    }
    
    // 4. 尝试标准化结果
    let normalizedResult: ExecutorAgentResult | undefined;
    if (errors.length === 0) {
      normalizedResult = {
        isNeedMcp: result.isNeedMcp,
        isTaskDown: result.isTaskDown,
        problem: result.problem,
        capabilityType: result.capabilityType,
        executionResult: result.executionResult
      };
    }
    
    return {
      isValid: errors.length === 0,
      errors,
      normalizedResult
    };
  }
  
  /**
   * 验证并修复（如果可以修复）
   */
  static validateAndRepair(result: any): {
    isValid: boolean;
    errors: string[];
    repairedResult?: ExecutorAgentResult;
  } {
    const validation = this.validate(result);
    
    if (validation.isValid) {
      return {
        isValid: true,
        errors: [],
        repairedResult: validation.normalizedResult
      };
    }
    
    // 尝试修复一些常见问题
    const repairedResult = this.tryRepair(result);
    if (repairedResult) {
      const revalidation = this.validate(repairedResult);
      if (revalidation.isValid) {
        return {
          isValid: true,
          errors: validation.errors, // 保留原始错误作为警告
          repairedResult: revalidation.normalizedResult
        };
      }
    }
    
    return {
      isValid: false,
      errors: validation.errors
    };
  }
  
  /**
   * 尝试修复常见问题
   */
  private static tryRepair(result: any): ExecutorAgentResult | null {
    if (!result) return null;
    
    // 修复 1：isNeedMcp 和 isTaskDown 都是 undefined
    if (result.isNeedMcp === undefined &amp;&amp; result.isTaskDown === undefined) {
      // 根据是否有 problem 来推断
      if (result.problem) {
        return {
          isNeedMcp: true,
          isTaskDown: false,
          problem: result.problem,
          capabilityType: result.capabilityType,
          executionResult: result.executionResult
        };
      } else if (result.executionResult) {
        return {
          isNeedMcp: false,
          isTaskDown: true,
          problem: result.problem,
          capabilityType: result.capabilityType,
          executionResult: result.executionResult
        };
      }
    }
    
    // 修复 2：isNeedMcp 是 string 类型
    if (typeof result.isNeedMcp === 'string') {
      return {
        ...result,
        isNeedMcp: result.isNeedMcp.toLowerCase() === 'true'
      };
    }
    
    // 修复 3：isTaskDown 是 string 类型
    if (typeof result.isTaskDown === 'string') {
      return {
        ...result,
        isTaskDown: result.isTaskDown.toLowerCase() === 'true'
      };
    }
    
    return null;
  }
}
```

### 3.2 简化三层硬编码检查

**目标**：保留验证，但让它更智能，或者作为 LLM 决策的参考，而不是硬性阻断。

**修改位置**：`src/lib/services/subtask-execution-engine.ts` 中的 `handlePreCompletedReview()`

**当前做法**：三层检查，任何一层不通过就直接打回 pending

**新做法**：将检查结果作为 Agent B 决策的参考，而不是硬性阻断

```typescript
private async handlePreCompletedReview(task: typeof agentSubTasks.$inferSelect) {
  console.log('[SubtaskEngine] Agent B: 评审 pre_completed 状态（执行Agent说搞定了）');
  
  // ==========================================
  // ✅ 改为：执行检查，但不直接阻断，而是将结果作为参考
  // ==========================================
  
  const checks = {
    hasValidResult: false,
    contentLengthValid: false,
    hasMatchingKeyword: false,
    details: [] as string[]
  };
  
  // 检查 1：execution_result 是否存在
  console.log('[SubtaskEngine] Agent B: [检查1/3] 检查 execution_result 是否存在...');
  checks.hasValidResult = task.executionResult &amp;&amp; task.executionResult !== 'null' &amp;&amp; task.executionResult !== '""';
  if (!checks.hasValidResult) {
    checks.details.push('execution_result 为 null 或空');
  }
  
  // 检查 2：内容长度
  console.log('[SubtaskEngine] Agent B: [检查2/3] 检查内容长度...');
  if (checks.hasValidResult) {
    let executionResultContent: string;
    try {
      const parsed = JSON.parse(task.executionResult);
      executionResultContent = typeof parsed === 'string' ? parsed : JSON.stringify(parsed);
    } catch (e) {
      executionResultContent = task.executionResult;
    }
    
    checks.contentLengthValid = executionResultContent.length &gt;= 20;
    if (!checks.contentLengthValid) {
      checks.details.push(`内容过短，仅 ${executionResultContent.length} 字符`);
    }
  }
  
  // 检查 3：关键词匹配
  console.log('[SubtaskEngine] Agent B: [检查3/3] 检查内容与任务是否基本相符...');
  if (checks.hasValidResult) {
    // ... 关键词匹配逻辑 ...
    checks.hasMatchingKeyword = true; // 简化
  }
  
  console.log('[SubtaskEngine] Agent B: 检查结果:', checks);
  
  // ==========================================
  // ✅ 关键改变：不再直接阻断，而是将检查结果传给 Agent B
  // ==========================================
  console.log('[SubtaskEngine] Agent B: 将检查结果作为参考，继续正常评审');
  await this.executeAgentBDecisionAndMcp(task, checks);
}
```

然后在 `executeAgentBDecisionAndMcp()` 中，将检查结果包含在给 Agent B 的提示词中。
